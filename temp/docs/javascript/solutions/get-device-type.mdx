## 使用 userAgent

function getDeviceType() {
  const userAgent = navigator.userAgent.toUpperCase();
  if (/IPHONE|IPOD/.test(userAgent) && /MOBILE/.test(userAgent)) {
    return "iPhone";

    return "Android";
